# Validator

### 基本使用
<Playground>
  <Validator validates={{
    required: true,
    num: {
      message: "请输入纯数字"
    },
    min: {
      value: 5,
      message: "最小为5"
    },
    custom: {
      rule: (val) => {
        return val.indexOf("1") > -1
      },
      message: "输入字符中必须包含1"
    }
  }}>
    <Input></Input>
  </Validator>
